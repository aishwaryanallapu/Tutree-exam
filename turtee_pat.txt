#include <iostream>
using namespace std;
int main() {
   for(int i=1;i<=7;i++){ //rows
       for(int k=1;k<=i;k++)   //spaces
           cout<<" ";
           for(int j=i;j<=7;j++)
           cout<<j<<" ";
           cout<<"\n";
   }
   //second triangle

   for(int i=6;i>=1;i--){
       for(int k=1;k<=i-1;k++){
       cout<<" ";}
     

       for(int t=i;t<=7;t++){
       cout<<t<<" ";
       }
       
       cout<<"\n";
return 0;
   }
